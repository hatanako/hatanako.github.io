``` csharp
using UnityEngine;

public class PlayerController : MonoBehaviour
{
    private Rigidbody2D rb;
    private Animator anim;

    public float moveSpeed = 5f;
    public float jumpForce = 5f;

    private bool isGrounded = true;
    public float groundCheckDistance = 0.1f;
    public Transform groundCheck;

    void Start()
    {
        rb = GetComponent<Rigidbody2D>();
        anim = GetComponentInChildren<Animator>();
    }

    void FixedUpdate()
    {
        // 使用 GetAxisRaw 获取离散输入（瞬间为 -1,0,1，无平滑）
        float inputArrow = Input.GetAxisRaw("Horizontal");
        Move(inputArrow);
    }


    private void Update()
    {
        CheckGround();
        Flip();
        Jump();
        CheckJumpState();

    }

    /// <summary>
    /// 检测地面
    /// </summary>
    private void CheckGround()
    {
        RaycastHit2D hit = Physics2D.Raycast(groundCheck.position, Vector2.down, groundCheckDistance, 1 << LayerMask.NameToLayer("Ground"));
        isGrounded = hit.collider != null;
        anim.SetBool("isGround", isGrounded);
    }

    private void OnDrawGizmosSelected()
    {
        // 仅在物体被选中时绘制
        Gizmos.color = isGrounded ? Color.green : Color.red;
        Gizmos.DrawLine(groundCheck.position, groundCheck.position + Vector3.down * groundCheckDistance);
    }

    /// <summary>
    /// 反转角色朝向，根据水平输入调整角色的 localScale.x 来实现翻转效果。
    /// </summary>
    private void Flip()
    {
        if (Input.GetAxisRaw("Horizontal") > 0)
        {
            transform.localScale = new Vector3(1, 1, 1);
        }
        else if (Input.GetAxisRaw("Horizontal") < 0)
        {
            transform.localScale = new Vector3(-1, 1, 1);
        }
    }

    /// <summary>
    /// 玩家移动方法，直接设置 Rigidbody2D 的 velocity 来实现移动。
    /// 根据输入方向调整水平速度，同时保留垂直速度（如重力和跳跃）。
    /// 动画状态根据实际速度更新，确保动画与移动同步。
    /// </summary>
    /// <param name="dir">输入的方向</param>
    private void Move(float dir)
    {
        // 直接设置速度，保留 y 轴速度（用于重力、跳跃等）
        rb.velocity = new Vector2(dir * moveSpeed, rb.velocity.y);

        // 根据实际速度（而非输入）更新动画，避免输入与移动不同步
        bool isMoving = Mathf.Abs(rb.velocity.x) > 0.01f;
        anim.SetBool("isMove", isMoving);
    }

    /// <summary>
    /// 跳跃方法
    /// </summary>
    private void Jump()
    {
        if(Input.GetKeyDown(KeyCode.Space) && isGrounded)
        {
            rb.velocity = new Vector2(rb.velocity.x, jumpForce);
        }
    }

    /// <summary>
    /// 检测玩家是在下降还是上升状态，更新动画参数以反映跳跃状态。
    /// </summary>
    private void CheckJumpState()
    {
        if(rb.velocity.y > 0.01f)
        {
            anim.SetBool("isJump", true);
            anim.SetBool("isFall", false);
        }
        else if(rb.velocity.y < -0.01f)
        {
            anim.SetBool("isJump", false);
            anim.SetBool("isFall", true);
        }
        else
        {
            anim.SetBool("isJump", false);
            anim.SetBool("isFall", false);
        }

    }
}
```

``` csharp
    public bool IsDashPressed() => Input.GetKeyDown(dash);
    public bool IsAttackPressed() => Input.GetKeyDown(attack);
    public bool IsCrouchPressed() => Input.GetKey(crouch);

}
```

首先我们需要一个类来管理这些状态，出于方便~~加上我觉得没必要~~，这里就直接用`PlayerController`作为中枢机构了，然后我们需要先创建一个状态机的基类：

``` csharp
public abstract class BaseState
{
    public abstract void Enter(PlayerController ctrl);
    public abstract void Update(PlayerController ctrl, InputManager input);
    public abstract void Exit(PlayerController ctrl);
}
```

这样我们所有需要实现的玩家状态机类只需要继承它，再在Update中写方法就好了，然后我们需要一个状态机切换的方法逻辑：

``` csharp
//PlayerController.cs
private BaseState nowState;

public void ChangeState(BaseState newState)
{
    nowState.Exit(this);
    nowState = newState;
    nowState.Enter(this);
}
```

```csharp
    public KeyCode leftMove = KeyCode.LeftArrow;

    public KeyCode dash = KeyCode.LeftShift;

    public bool IsDashPressed() => Input.GetKeyDown(dash);
    public bool IsAttackPressed() => Input.GetKeyDown(attack);
```
