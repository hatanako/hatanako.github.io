public void Dash(Transform transform)
{
    dashTimer = 0f;

    slideTimer = 0f;
    GCDTimer = GCD; // 重置全局冷却时间
    float faceDir = transform.localScale.x; // 获取当前角色的朝向
    float dir = input.IsLeftMovePressed() ? -1 : (input.IsRightMovePressed() ? 1 : faceDir); // 获取当前角色的朝向
    Flip(transform, dir);
    rb.velocity = new Vector2(dir * slideSpeed, 0);
}

public void KeepSlide()
{
    rb.velocity = new Vector2(slideSpeed * transform.localScale.x, rb.velocity.y);
}

using UnityEngine;

public class PlayerDashState : BaseState
{
    private float dashTime = 0.2f;
    public override void Enter(PlayerController ctrl)
    {
        ctrl.anim.SetBool("isDash",true);
        ctrl.move.Dash(ctrl.transform);
        dashTime = ctrl.move.dashDuration;
    }

    public override void Exit(PlayerController ctrl)
    {
        ctrl.anim.SetBool("isDash",false);
    }

    public override void Update(PlayerController ctrl, InputManager input)
    {
        dashTime -= Time.deltaTime;
        ctrl.move.KeepDash();
        if (dashTime <= 0)
        {
            if (!ctrl.IsGrounded)
                ctrl.ChangeState(ctrl.fallState);
            else
                ctrl.ChangeState(ctrl.idleState);
        }
    }
}

~~地狱绘图.jpg~~，所以我们得先想想办法把动画状态机给更改一下，我个人的思路是，我们可以写一个枚举和一个管理动画状态机的类：这里命名为`PlayerAnimController`，枚举中有各个状态机的名称，然后在切换代码状态机时去调用`PlayerAnimController`类里的切换状态逻辑，这样我们就只需要一个参数就可以切换所有的动画了。说做就做：我们先创建一个`ENUM`：

``` csharp
//PlayerAnimController.cs

/// <summary>
/// 状态机枚举
/// </summary>
public enum PlayerAnimState
{
    Idle = 0,
    Move = 1,
    Jump = 2,
    Fall = 3,
    Dash = 4,
    Crouch = 5,
    Slide = 6,
    WallSlide = 7,
    // 其他状态可以继续添加
}

/// <summary>
/// 动画控制器，负责根据状态机的状态切换动画
/// </summary>
public class PlayerAnimController
{
    private int nowState = 0;

    private PlayerAnimState state;

    public void SetAnimState(PlayerAnimState newState,Animator anim)
    {
        if (state == newState) return;
        state = newState;
        nowState = (int)state;
        anim.SetInteger("nowState", nowState);
    }
}
```

在这个状态下把左右键一起按下会出现这种卡住的效果（），我们先来解决这个问题，  
有一个非常暴力的做法就是进入滑墙状态会有cd，必须要待够一定时间才能切换成别的状态机，但是一款类银最不可或缺的就是一个快速响应的操作手感，所以是不可取的。  
那么还有什么方法可以解决呢？~~我也不知道~~，我们可以直接进行检测输入，如果是左右键一起按直接被检测为不合法的输入，不做任何事情：

``` csharp
//InputManager.cs
public bool IsLeftRightPressed() => Input.GetKey(leftMove) && Input.GetKey(rightMove);

//PlayerWallSlideState.cs
public override void Update(PlayerController ctrl, InputManager input)
{
    ctrl.jump.WallSlide();
    if (input.IsLeftRightPressed())
        return;
    if (ctrl.IsGrounded)
    {
        ctrl.ChangeState(ctrl.idleState);
    }
    else if(input.IsJumpPressed())
    {
        ctrl.ChangeState(ctrl.jumpState);
    }
     else if(!ctrl.IsTouchingWall || 
        (input.IsLeftMovePressed() && ctrl.faceDir == 1 || input.IsRightMovePressed() && ctrl.faceDir == -1))
    {
        ctrl.ChangeState(ctrl.fallState);
    }
}
```

一定要注意的是：WallSlide方法一定要放在检测按键之前，不然可能会出现左右键一起按直接卡着不动了~~天哪是90°光滑平面攀岩高手~~

现在基本上全都解决了，我们可以来做蹬墙跳的功能了，简单来说，就是在WallSlide状态中，按一下Jump按钮，我们就可以朝着对向方向跳了  
要注意：是对向方向噢，因为我们的WallSlide动画是反过来的，如果是在左手的墙，我们WallSlide看起来面向的是右边，实际的朝向还是左边。  
同样的还有一个问题，在进入跳跃的一瞬间，我们的射线会马上检测到墙体，然后马上又变成WallSlide状态，这里就需要我们动一下脑子了  

我们依旧可以做一个计时器来限制，但是考虑到快速响应问题，我们不可以在进入WallSlide的时候做这个逻辑，我们可以在PlayerJumper中加一个参数，通过这个参数来判断这次跳跃是否是蹬墙跳，然后在跳出去的时候加一个Timer，避免玩家马上返回，从而直接一路直跳上去。

``` csharp
//PlayerJumper.cs
public float wallJumpHorizontalMultiplier = 8f;
public bool isWallJump = false;

/// <summary>
/// 蹬墙跳方法
/// </summary>
/// <param name="faceDir">当前面朝向</param>
public void WallJump(float faceDir)
{
    rb.velocity = new Vector2(-faceDir * wallJumpHorizontalMultiplier, jumpForce);
    isWallJump = false;
}

//PlayerJumpState.cs
private float wallJumpTimer = 0.2f; // 蹬墙跳的持续时间
public override void Enter(PlayerController ctrl)
{
    if(!ctrl.jump.isWallJump)
        ctrl.jump.Jump(ctrl.IsGrounded);
    else
    {
        wallJumpTimer = 0.2f; // 重置蹬墙跳的持续时间
        ctrl.jump.WallJump(ctrl.faceDir);
    }
    ctrl.stateType = PlayerAnimState.Jump;
}

public override void Update(PlayerController ctrl, InputManager input)
{
    wallJumpTimer -= Time.deltaTime;
    if (wallJumpTimer > 0)
        return;
    //其他代码
}
```
